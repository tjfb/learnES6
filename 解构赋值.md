# 解构赋值的规则是，只要等号右边的值不是对象或数组，就先将其转为对象。由于undefined和null无法转为对象，所以对它们进行解构赋值，都会报错。
`
let { prop: x } = undefined; // TypeError
let { prop: y } = null; // TypeError
`

# 用途
## 1 交换变量
`
let x = 1;
let y = 2;

[x, y] = [y, x];
`
## 2 从函数返回多个值
`
// 返回一个数组

function example() {
  return [1, 2, 3];
}
let [a, b, c] = example();

// 返回一个对象

function example() {
  return {
    foo: 1,
    bar: 2
  };
}
let { foo, bar } = example();
`
